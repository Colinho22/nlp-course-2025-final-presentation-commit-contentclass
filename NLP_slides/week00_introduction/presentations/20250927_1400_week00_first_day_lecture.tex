% Slide 17: The Transformer Revolution
\begin{frame}{Understanding the AI Revolution}
\begin{center}
{\Large \textbf{Why Transformers Changed Everything}}
\end{center}

\vspace{5mm}

\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Before Transformers (Pre-2017):}
\begin{itemize}
\item Sequential processing (RNNs)
\item Slow training (hours to days)
\item Limited context (hundreds of words)
\item Vanishing gradient problems
\item Hard to parallelize
\item Specialized architectures per task
\end{itemize}

\vspace{3mm}
\textbf{Limitations We Hit:}
\begin{itemize}
\item Couldn't scale to large models

\item Parallel processing (attention)
\item Fast training (hours on GPUs)
\item Unlimited context (thousands of tokens)
\item Stable gradients
\item Massively parallelizable
\item Universal architecture
\end{itemize}

\vspace{3mm}
\textbf{What Became Possible:}
\begin{itemize}
\item GPT-3: 175 billion parameters
\item GPT-4: Multimodal understanding
\item Claude: 100k+ token context
\item ChatGPT: Conversational AI
\item All modern LLMs use transformers
\end{itemize}
