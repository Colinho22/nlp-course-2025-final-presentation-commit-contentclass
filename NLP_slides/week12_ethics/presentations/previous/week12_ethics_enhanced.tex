% Bias in language models
\begin{frame}[t]{Bias in Language Models: Mirror of Society}
    \textbf{Where bias comes from:}
    
    \vspace{0.5em}
    \centering
    \includegraphics[width=0.65\textwidth]{../figures/bias_sources.pdf}
    
    \vspace{0.5em}
    \textbf{Types of bias:}\footnotemark
    \begin{itemize}
        \item \textbf{Historical bias}: Past discrimination in data
        \item \textbf{Representation bias}: Underrepresented groups
        \item \textbf{Measurement bias}: What we choose to measure
        \item \textbf{Aggregation bias}: One-size-fits-all models
        \item \textbf{Evaluation bias}: Biased benchmarks
    \end{itemize}
    
    \footnotetext{Suresh \& Guttag (2021) "A Framework for Understanding Sources of Harm"}
\end{frame}

% Measuring bias
\begin{frame}[t]{Detecting and Measuring Bias}
    \textbf{Key approaches to detect bias:}
    
    \vspace{0.5em}
    \textbf{1. Template-based testing:}
    \begin{itemize}
        \item Fill-in-the-blank: "The [MASK] is a doctor"
        \item Compare male vs female completion rates
        \item Measure occupation stereotypes systematically
    \end{itemize}
    
    \vspace{0.5em}
    \textbf{2. Word Embedding Association Test (WEAT):}
    \begin{itemize}
        \item Compare semantic associations between groups
        \item Measure implicit biases in word embeddings
        \item Statistical significance testing for bias
    \end{itemize}
    
    \vspace{0.5em}
    \textbf{3. Counterfactual evaluation:}
    \begin{itemize}
        \item Swap demographic attributes in text
        \item Measure prediction differences
        \item Identify systematic discriminatory patterns
    \end{itemize}
    
    \vspace{0.5em}
    \textbf{Common findings across models:}
    \begin{itemize}
        \item Gender: 3:1 male bias in technical roles
        \item Race: Name-based discrimination in hiring contexts
        \item Age: Strong preference for youth-associated terms
        \item Toxicity: Small percentage but harmful impact
    \end{itemize}
\end{frame}

% Privacy and security
\begin{frame}[t]{Privacy and Security: Protecting User Data}
    \textbf{Privacy risks in language models:}\footnotemark
    
    \vspace{0.5em}
    \begin{itemize}
        \item \textbf{Memorization}: Models can leak training data
        \item \textbf{Inference attacks}: Extract personal information
        \item \textbf{Re-identification}: Deanonymize text
        \item \textbf{Model inversion}: Reconstruct training examples
    \end{itemize}
    
    \vspace{0.5em}
    \textbf{Protection techniques:}
    \begin{itemize}
        \item Differential privacy training
        \item Federated learning
        \item Secure multi-party computation
        \item Data minimization
        \item Regular audits
    \end{itemize}
    
    \vspace{0.5em}
    \centering
    \includegraphics[width=0.5\textwidth]{../figures/privacy_protection.pdf}
    
    \footnotetext{Carlini et al. (2021) "Extracting Training Data from Large Language Models"}
\end{frame}
