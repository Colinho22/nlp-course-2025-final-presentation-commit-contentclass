% Mathematical Content
\begin{frame}{Mathematical Formulas}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Attention Mechanism}
\begin{align*}
\text{score}(h_t, \bar{h}_s) &= h_t^T W_a \bar{h}_s\\
\alpha_{ts} &= \frac{\exp(\text{score}(h_t, \bar{h}_s))}{\sum_{s'} \exp(\text{score}(h_t, \bar{h}_{s'}))}\\
c_t &= \sum_s \alpha_{ts} \cdot \bar{h}_s
\end{align*}

\column{0.48\textwidth}
\textbf{Key Components}
\begin{itemize}
\item $h_t$ - \secondary{decoder hidden state}
\item $\bar{h}_s$ - \secondary{encoder hidden states}
\item $\alpha_{ts}$ - \highlight{attention weights}
\item $c_t$ - \secondary{context vector}
\end{itemize}
\end{columns}

% Code Example
\begin{frame}[fragile]{Code Highlighting}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Python Implementation}
\begin{verbatim}
def attention(query, keys, values):
    # Calculate scores
    scores = torch.matmul(
        query, keys.transpose(-2, -1)
    )
    # Apply softmax
    weights = F.softmax(scores, dim=-1)
    # Weighted sum
    output = torch.matmul(weights, values)
    return output, weights
\end{verbatim}

\column{0.48\textwidth}
\textbf{Key Points}
\begin{itemize}
\item \highlight{Query} - what we're looking for
\item \info{Keys} - what we compare against
\item \success{Values} - what we aggregate
\item \secondary{Weights} - attention distribution
\end{itemize}
