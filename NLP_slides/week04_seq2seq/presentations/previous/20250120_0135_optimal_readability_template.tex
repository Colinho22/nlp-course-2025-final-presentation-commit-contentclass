\textbf{\large Python Implementation}
\begin{verbatim}
def attention(Q, K, V):
    # Compute attention scores
    scores = torch.matmul(Q, K.T)
    scores = scores / sqrt(d_k)

    # Apply softmax
    weights = F.softmax(scores, dim=-1)

    # Weighted sum of values
    output = torch.matmul(weights, V)
    return output, weights
\end{verbatim}

\column{0.48\textwidth}
\textbf{\large Key Concepts}
\begin{itemize}
\item \highlight{Q}: Query matrix
\item \highlight{K}: Key matrix
\item \highlight{V}: Value matrix
\item \data{d\_k}: Key dimension
\item Scores: $QK^T / \sqrt{d_k}$
\item Weights: $\text{softmax}(\text{scores})$
\item Output: Weighted values
\end{itemize}
